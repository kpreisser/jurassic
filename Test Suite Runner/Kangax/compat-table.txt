To generate
===========

Go to http://kangax.github.io/compat-table/es6/

The JSON is generated by running this script (in the console):

var scripts = document.querySelectorAll("script[data-source]");
var results = [];
for (var i in scripts) {
  if (scripts[i].attributes && scripts[i].attributes["data-source"]) {
    var result = {};
    var parentIdAttr = scripts[i].parentElement.parentElement.attributes["data-parent"];
	var parentCell;
    if (parentIdAttr)
      parentCell = document.getElementById("test-" + parentIdAttr.value);
	else
	  parentCell = scripts[i].parentElement;
    var groupElement = parentCell.parentElement;
	while (groupElement != null && groupElement.className != "category") {
	  groupElement = groupElement.previousElementSibling;
    }

	result.name = parentCell.innerText.trim();
	if (result.name[0] == '§')
	  result.name = result.name.substr(1).trim();
    if (result.name.substr(-1) === "►" || result.name.substr(-1) === "c")
      result.name = result.name.substr(0, result.name.length - 1).trim();

    result.group = groupElement.firstElementChild.innerText.trim();

    result.detail = scripts[i].previousElementSibling.innerText.trim();
	if (result.detail[0] == '§')
	  result.detail = result.detail.substr(1).trim();

    result.script = scripts[i].attributes["data-source"].value;
    results.push(result);
  }
}
document.body.innerText = JSON.stringify(results);
"success";
